需求: 获取父级部门的直接子部门, 按照PRIORITY排序, 如果相同, 则按照ENTITY_PINYIN表中的全拼FULLPY排序;

SELECT A.DEPARTID,A.DEPARTNAME,A.PRIORITY FROM (SELECT DEPARTID,DEPARTNAME,PRIORITY FROM DEPARTMENT WHERE PARENTID=0 ORDER BY PRIORITY) A LEFT JOIN ENTITY_PINYIN WHERE A.DEPARTID=ENTITY_PINYIN.[ENTITYID] ORDER BY PRIORITY ASC,FULLPY ASC;

SELECT DEPARTID,DEPARTNAME,PRIORITY FROM (SELECT DEPARTID,DEPARTNAME,PRIORITY,FULLPY,PARENTID FROM DEPARTMENT LEFT JOIN ENTITY_PINYIN WHERE DEPARTID=ENTITYID) WHERE PARENTID=0 ORDER BY PRIORITY ASC,FULLPY ASC;
